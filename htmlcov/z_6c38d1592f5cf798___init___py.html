1"""Loxo dlt source — people, jobs, companies, deals, activities, users. 

2 

3Spec: https://loxo.readme.io/reference/loxo-api 

4 

5Auth: Bearer API token. Base URL is per-agency — 

6`https://{domain}/api/{agency_slug}/...` — but `domain` is `app.loxo.co` 

7for nearly every customer, so only two values are per-customer secrets: 

8`agency_slug` and `api_key`. Override `domain` via the `LOXO_API_BASE_URL` 

9env var for the rare custom-domain agency. 

10 

11Pagination is mixed: 

12- `scroll_id` (cursor) — people, companies, deals, activities. Pass the 

13 `scroll_id` from the response back as a query param on the next call. 

14- `page` / `per_page` — jobs only. Stock `PageNumberPaginator`. 

15 

16# OPEN QUESTIONS — settle these with a sandbox key before locking design 

17 

181. **No documented `updated_at` filter.** Listing endpoints don't appear 

19 to support a `since=` / `updated_at_from=` parameter, so every resource 

20 here is `write_disposition="replace"` for now. If a filter DOES exist 

21 in practice, switch to `append` + `dlt.sources.incremental("updated_at", 

22 initial_value=EPOCH_ISO, range_start="open")` and pass 

23 `cursor.start_value` (NOT `last_value`) to the API param — per 

24 PAR-116 trap memo on early-termination patterns. 

25 

262. **`scroll_id` stability across runs.** If the cursor is session-bound 

27 (Elastic-style) it cannot be persisted across Dagster runs, which 

28 pins us to full re-list each run. If it's a stable opaque token, 

29 we can resume mid-scan — relevant only once incremental sync is on. 

30 

31Resources: 

32- `people`, `companies`, `deals`, `activities` — scroll_id, replace. 

33 The `activities` resource hits Loxo's `/person_events` endpoint 

34 (Loxo's URL naming; we keep the friendlier dlt name). 

35- `jobs` — page-based, replace. 

36- `users` — single-page list (no pagination metadata documented), replace. 

37"""

67@dlt.source(name="loxo") 

68def loxo_source( 

69 agency_slug: str = dlt.config.value, 

70 api_key: str = dlt.secrets.value, 

71 domain: str = _DEFAULT_DOMAIN, 

72 base_url: str | None = None, 

73) -> list[Any]: 

74 """Loxo source factory — yields six resources covering ATS/CRM data. 

75 

76 Args: 

77 agency_slug: Per-agency URL segment (``https://{domain}/api/{agency_slug}/...``). 

78 api_key: Loxo API token. Resolved from secrets by default. 

79 domain: Loxo host. Defaults to ``app.loxo.co``; rarely overridden. 

80 base_url: Test seam — pass the full pre-built base URL to bypass 

81 ``_base_url`` construction (avoids env-var leakage between 

82 parallel tests). Production calls leave this ``None``. 

83 """ 

84 # Test seam: tests pass `base_url=` to bypass `_base_url` construction. 

85 # Production callers leave `base_url=None`, which routes through `_client` 

86 # → `_base_url` and the optional `LOXO_API_BASE_URL` env override. 

87 if base_url is not None: 

88 client = RESTClient( 

89 base_url=base_url, 

90 auth=BearerTokenAuth(api_key), 

91 headers={"Accept": "application/json"}, 

92 ) 

93 else: 

94 client = _client(domain, agency_slug, api_key) 

95 

96 def _scroll(path: str, data_selector: str) -> Iterator[Row]: 

97 # Loxo's `/companies` and `/deals` reject `per_page` with 422 

98 # ("Invalid parameters: [:per_page]") even though `/people` and 

99 # `/person_events` accept it. Inconsistent per-endpoint validation — 

100 # safest is to omit and let scroll_id pagination handle volume via 

101 # Loxo's server-side default page size. 

102 for page in client.paginate( 

103 path, 

104 paginator=LoxoScrollIdPaginator(), 

105 data_selector=data_selector, 

106 ): 

107 yield from page 

108 

109 @dlt.resource(name="people", primary_key="id", write_disposition="replace") 

110 def people() -> Iterator[Row]: 

111 yield from _scroll(ENDPOINT_PEOPLE, data_selector="people") 

112 

113 @dlt.resource(name="jobs", primary_key="id", write_disposition="replace") 

114 def jobs() -> Iterator[Row]: 

115 for page in client.paginate( 

116 ENDPOINT_JOBS, 

117 params={"per_page": JOBS_PAGE_SIZE}, 

118 paginator=PageNumberPaginator(base_page=1), 

119 data_selector="jobs", 

120 ): 

121 yield from page 

122 

123 @dlt.resource(name="companies", primary_key="id", write_disposition="replace") 

124 def companies() -> Iterator[Row]: 

125 yield from _scroll(ENDPOINT_COMPANIES, data_selector="companies") 

126 

127 @dlt.resource(name="deals", primary_key="id", write_disposition="replace") 

128 def deals() -> Iterator[Row]: 

129 yield from _scroll(ENDPOINT_DEALS, data_selector="deals") 

130 

131 @dlt.resource(name="activities", primary_key="id", write_disposition="replace") 

132 def activities() -> Iterator[Row]: 

133 # Loxo's API names this collection `person_events` (returns 404 on 

134 # `/activities`). The dlt resource keeps the friendlier name for the 

135 # downstream warehouse table. 

136 yield from _scroll(ENDPOINT_PERSON_EVENTS, data_selector="person_events") 

137 

138 @dlt.resource(name="users", primary_key="id", write_disposition="replace") 

139 def users() -> Iterator[Row]: 

140 for page in client.paginate( 

141 ENDPOINT_USERS, 

142 paginator=SinglePagePaginator(), 

143 data_selector="users", 

144 ): 

145 yield from page 

146 

147 return [people, jobs, companies, deals, activities, users]
