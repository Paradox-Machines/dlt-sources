1"""Microsoft 365 / Graph source — endpoint constants and API configuration. 

2 

3All Graph paths are versioned under ``/v1.0``. The mail plane reads from a 

4single **shared mailbox** (``ndas@point41.com``) via the application-permission 

5``/users/{mailbox}`` addressing scheme, and the document plane reads from a 

6**SharePoint document library** exposed as a drive under ``/sites/{site_id}``. 

7""" 

8 

9from __future__ import annotations 

10 

11# --- Microsoft identity platform (OAuth 2.0 client-credentials) -------------- 

12# Override the login host in tests via the M365_GRAPH_LOGIN_BASE_URL env var. 

13LOGIN_BASE_URL = "https://login.microsoftonline.com" 

14 

15# The v2.0 token endpoint for a given tenant. Formatted with the tenant id 

16# (GUID or verified domain) at client construction time. 

17PATH_TOKEN = "/{tenant_id}/oauth2/v2.0/token" 

18 

19# App-only flows request the static ``.default`` scope: the union of the 

20# application permissions an admin has consented to for the app registration. 

21DEFAULT_SCOPE = "https://graph.microsoft.com/.default" 

22 

23# Refresh the cached access token this many seconds before its stated expiry 

24# to avoid racing Graph's ~3600s TTL on long extracts. 

25TOKEN_REFRESH_SAFETY_SECONDS = 120 

26 

27# --- Microsoft Graph --------------------------------------------------------- 

28# Override the Graph host in tests via the M365_GRAPH_API_BASE_URL env var. 

29GRAPH_API_BASE_URL = "https://graph.microsoft.com" 

30GRAPH_API_VERSION = "v1.0" 

31 

32# Mail plane. ``mailbox`` is the shared mailbox UPN (e.g. ndas@point41.com); 

33# application permissions let us address it directly under /users/{mailbox}. 

34# The shared NDA intake mailbox — overridable via M365_GRAPH_MAILBOX env var. 

35DEFAULT_MAILBOX = "ndas@point41.com" 

36PATH_MESSAGES = "/{version}/users/{mailbox}/messages" 

37PATH_MESSAGE_ATTACHMENTS = "/{version}/users/{mailbox}/messages/{message_id}/attachments" 

38 

39# Document plane. A SharePoint document library is the default drive of a site. 

40PATH_DRIVE_CHILDREN = "/{version}/sites/{site_id}/drive/root/children" 

41PATH_DRIVE_ITEM_CONTENT = "/{version}/sites/{site_id}/drive/items/{item_id}/content" 

42 

43# Graph's ``@odata.type`` discriminator for file (vs item / reference) attachments. 

44FILE_ATTACHMENT_ODATA_TYPE = "#microsoft.graph.fileAttachment" 

45 

46# Incremental cursor floor — the Unix epoch as an RFC-3339 string. Graph 

47# ``receivedDateTime`` / ``lastModifiedDateTime`` values are RFC-3339 with a 

48# ``Z`` offset, so lexicographic comparison orders them correctly. 

49EPOCH_ISO = "1970-01-01T00:00:00Z" 

50 

51# Page size for the mail list endpoint (Graph caps $top at 1000 for messages). 

52MESSAGES_PAGE_SIZE = 50 

53 

54# --- Classification vocabulary ---------------------------------------------- 

55# Document roles emitted on the `doc_role` column of the dropzone table. 

56ROLE_FRESH_NDA = "fresh_nda" 

57ROLE_COUNTERPARTY_RETURN = "counterparty_return" 

58ROLE_TEASER = "teaser" 

59 

60# Source-system tags on the `source_system` column. 

61SYSTEM_OUTLOOK = "outlook" 

62SYSTEM_SHAREPOINT = "sharepoint" 

63 

64# Filename tokens that mark an attachment/file as a deal teaser rather than an 

65# NDA itself. A fresh NDA outbound "often ships with a teaser" — the teaser is 

66# classified separately so the dropzone poller does not treat it as an NDA. 

67TEASER_TOKENS = ( 

68 "teaser", 

69 "teaser-deck", 

70 "cim", 

71 "one-pager", 

72 "onepager", 

73 "flyer", 

74 "overview", 

75) 

76 

77# Filename tokens that mark a SharePoint file as a counterparty return (a marked 

78# up / redlined / countersigned copy) rather than the fresh outbound draft. 

79# The mail plane classifies by thread position instead; these are the fallback 

80# signal for the document plane, which has no conversation context. 

81RETURN_TOKENS = ( 

82 "return", 

83 "returned", 

84 "redline", 

85 "redlined", 

86 "markup", 

87 "marked-up", 

88 "markedup", 

89 "counter", 

90 "countersigned", 

91 "revised", 

92 "comments", 

93 "tracked", 

94) 

95 

96# HTTP status codes with special handling. 

97HTTP_NOT_FOUND = 404
